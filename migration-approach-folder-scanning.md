    # Migration Approach: Upload-Based → Folder-Scanning Music Player

## Overview

Transform the current upload-based music player into a folder-scanning system that automatically organizes music from user's local file system.

---

## Current System vs Target System

### Current (What We Built):
```
User → Upload files → IndexedDB stores blobs → Play from IndexedDB
```

### Target (What You Want):
```
User → Select Music folder → Scan recursively → Store file references → Play from original files
```

### 1.2 Create Folder Scanner Utility

**New File:** `src/lib/filesystem/folderScanner.js`

**Purpose:** Recursively scan folder and extract all audio files

```javascript
export async function scanMusicFolder(dirHandle) {
  const songs = [];
  const playlists = new Map(); // folder name → songs
  
  // Recursive function to scan directory
  async function scanDirectory(dirHandle, path = '') {
    for await (const entry of dirHandle.values()) {
      
      if (entry.kind === 'file') {
        // Check if audio file
        if (isAudioFile(entry.name)) {
          const file = await entry.getFile();
          const metadata = await extractMetadata(file);
          
          // Create song object
          const song = {
            ...metadata,
            filePath: `${path}/${entry.name}`,
            fileHandle: entry, // Store file handle (important!)
            folderName: getFolderName(path) // For playlist generation
          };
          
          songs.push(song);
          
          // Add to playlist based on folder
          const playlistName = getFolderName(path) || 'All Songs';
          if (!playlists.has(playlistName)) {
            playlists.set(playlistName, []);
          }
          playlists.get(playlistName).push(song);
        }
      } 
      else if (entry.kind === 'directory') {
        // Recursively scan subdirectory
        await scanDirectory(entry, `${path}/${entry.name}`);
      }
    }
  }
  
  await scanDirectory(dirHandle);
  
  return { songs, playlists };
}

function isAudioFile(filename) {
  const audioExtensions = ['.mp3', '.wav', '.flac', '.m4a', '.ogg'];
  return audioExtensions.some(ext => filename.toLowerCase().endsWith(ext));
}

function getFolderName(path) {
  // Extract last folder name from path
  const parts = path.split('/').filter(Boolean);
  return parts[parts.length - 1] || 'Root';
}
```

**Key Logic:**
- Uses `for await...of` to iterate folder entries
- Recursively enters subdirectories
- Filters audio files by extension
- Groups songs by folder for playlists
- Stores **file handle** not file blob

```javascript
// Store root folder handle for re-access
export async function storeFolderHandle(dirHandle) {
  await db.folderHandle.put({
    id: 'music-folder',
    handle: dirHandle,
    grantedAt: new Date()
  });
}

// Get folder handle (for returning users)
export async function getFolderHandle() {
  const record = await db.folderHandle.get('music-folder');
  
  if (record) {
    // Check if permission still granted
    const permission = await record.handle.queryPermission({ mode: 'read' });
    
    if (permission === 'granted') {
      return record.handle;
    } else {
      // Request permission again
      const newPermission = await record.handle.requestPermission({ mode: 'read' });
      if (newPermission === 'granted') {
        return record.handle;
      }
    }
  }
  
  return null;
}

// Store file handle for each song
export async function storeFileHandle(filePath, fileHandle) {
  await db.fileHandles.put({
    filePath,
    handle: fileHandle
  });
}

// Get file from handle (for playback)
export async function getAudioFile(filePath) {
  const record = await db.fileHandles.get(filePath);
  if (!record) return null;
  
  try {
    const file = await record.handle.getFile();
    return file;
  } catch (error) {
    console.error('Error accessing file:', error);
    return null;
  }
}
```

**Important:**
- **File handles** can be stored in IndexedDB (new browser feature!)
- Handles persist across sessions (with permission)
- Must check/request permission on each access

## Phase 2: Auto-Generate Playlists from Folders

### 2.1 Create Auto-Playlist Generator

**New File:** `src/lib/playlists/autoPlaylistGenerator.js`

```javascript
export function generatePlaylistsFromFolders(songs) {
  const playlistMap = new Map();
  
  songs.forEach(song => {
    const folderName = song.folderName || 'Uncategorized';
    
    if (!playlistMap.has(folderName)) {
      playlistMap.set(folderName, {
        name: folderName,
        songs: [],
        isAutoGenerated: true,
        source: 'folder',
        createdAt: new Date()
      });
    }
    
    playlistMap.get(folderName).songs.push(song.id);
  });
  
  return Array.from(playlistMap.values());
}
```

**Example:**
```
Music/
  Gym/
    track1.mp3 → folderName: "Gym"
    track2.mp3 → folderName: "Gym"
  Chill/
    track3.mp3 → folderName: "Chill"

Result:
[
  { name: "Gym", songs: [1, 2], isAutoGenerated: true },
  { name: "Chill", songs: [3], isAutoGenerated: true }
]
```

```javascript
export const usePlaylistStore = create((set) => ({
  // ... existing state
  
  // NEW: Auto-generate playlists
  generateFromFolders: async (songs) => {
    const autoPlaylists = generatePlaylistsFromFolders(songs);
    
    // Save to IndexedDB
    for (const playlist of autoPlaylists) {
      await db.playlists.add(playlist);
    }
    
    set({ playlists: autoPlaylists });
  },
  
  // NEW: Mark playlist as user-created vs auto-generated
  createUserPlaylist: async (name) => {
    const playlist = {
      name,
      songs: [],
      isAutoGenerated: false, // User created
      createdAt: new Date()
    };
    
    const id = await db.playlists.add(playlist);
    set((state) => ({
      playlists: [...state.playlists, { ...playlist, id }]
    }));
  }
}));
```

## Phase 4: Implement Re-Scan on Return

### 4.1 Create Sync Logic

**New File:** `src/lib/filesystem/syncLibrary.js`

```javascript
export async function syncLibrary() {
  // Get stored folder handle
  const dirHandle = await getFolderHandle();
  
  if (!dirHandle) {
    console.log('No folder access, user needs to select folder');
    return { needsSetup: true };
  }
  
  // Scan current state of folder
  const { songs: currentSongs } = await scanMusicFolder(dirHandle);
  
  // Get existing songs from IndexedDB
  const existingSongs = await db.songs.toArray();
  
  // Find new songs (in folder but not in DB)
  const newSongs = currentSongs.filter(current => 
    !existingSongs.some(existing => existing.filePath === current.filePath)
  );
  
  // Find deleted songs (in DB but not in folder)
  const deletedSongs = existingSongs.filter(existing =>
    !currentSongs.some(current => current.filePath === existing.filePath)
  );
  
  // Add new songs to DB
  for (const song of newSongs) {
    await db.songs.add(song);
    await storeFileHandle(song.filePath, song.fileHandle);
  }
  
  // Remove deleted songs from DB
  for (const song of deletedSongs) {
    await db.songs.delete(song.id);
    await db.fileHandles.delete(song.filePath);
  }
  
  console.log(`Sync complete: +${newSongs.length} songs, -${deletedSongs.length} songs`);
  
  return {
    needsSetup: false,
    added: newSongs.length,
    removed: deletedSongs.length
  };
}
```

### 4.2 Call Sync on App Load

**File:** `src/App.jsx`

```javascript
function App() {
  const [isLoading, setIsLoading] = useState(true);
  const [needsSetup, setNeedsSetup] = useState(false);
  
  useEffect(() => {
    async function initializeApp() {
      setIsLoading(true);
      
      // Try to sync library with folder
      const result = await syncLibrary();
      
      if (result.needsSetup) {
        // First-time user, show folder selection
        setNeedsSetup(true);
      } else {
        // Existing user, library synced
        await fetchSongs();
        await fetchPlaylists();
      }
      
      setIsLoading(false);
    }
    
    initializeApp();
  }, []);
  
  if (isLoading) return <LoadingScreen />;
  if (needsSetup) return <FolderSetupScreen />;
  
  return (
    // ... normal app
  );
}
```

```jsx
export default function FolderSetupScreen() {
  const [isScanning, setIsScanning] = useState(false);

    setIsScanning(true);
    
    try {
      const dirHandle = await window.showDirectoryPicker();
```
